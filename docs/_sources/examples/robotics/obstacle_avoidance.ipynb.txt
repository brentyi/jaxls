    "class WaypointVar(jaxls.Var[jax.Array], default_factory=lambda: jnp.zeros(2)):\n",
    "    \"\"\"2D waypoint position.\"\"\"\n",
    "\n",
    "\n",
    "# Batched variable creation.\n",
    "waypoint_vars = WaypointVar(id=jnp.arange(n_waypoints))"
   ]

  {
   "cell_type": "code",
   "execution_count": 5,

    "@jaxls.Cost.factory\n",
    "def smoothness_cost(\n",
    "    vals: jaxls.VarValues,\n",
    "    var_prev: WaypointVar,\n",
    "    var_curr: WaypointVar,\n",
    "    var_next: WaypointVar,\n",
    ") -> jax.Array:\n",
    "    \"\"\"Penalize acceleration (second derivative).\"\"\"\n",
    "    accel = vals[var_prev] - 2 * vals[var_curr] + vals[var_next]\n",
    "    return accel * 5.0  # Weight for smoothness\n",
    "\n",
    "\n",
    "@jaxls.Cost.factory\n",
    "def anchor_cost(\n",
    "    vals: jaxls.VarValues,\n",
    "    var: WaypointVar,\n",
    "    target: jax.Array,\n",
    ") -> jax.Array:\n",
    "    \"\"\"Pin waypoint to target position.\"\"\"\n",
    "    return (vals[var] - target) * 10.0  # Strong weight\n",
    "\n",
    "\n",
    "@jaxls.Cost.factory(kind=\"constraint_leq_zero\")\n",
    "def obstacle_constraint(\n",
    "    vals: jaxls.VarValues,\n",
    "    var: WaypointVar,\n",
    "    center: jax.Array,\n",
    "    radius: float,\n",
    ") -> jax.Array:\n",
    "    \"\"\"Stay outside obstacle: ||p - center|| >= radius.\"\"\"\n",
    "    diff = vals[var] - center\n",
    "    dist = jnp.sqrt(jnp.sum(diff**2) + 1e-6)\n",
    "    # Constraint: radius - dist <= 0 (i.e., dist >= radius)\n",
    "    return jnp.array([radius - dist])"
   ]

    "# Indices for smoothness costs (interior waypoints)\n",
    "smooth_prev = jnp.arange(0, n_waypoints - 2)  # 0, 1, ..., n-3\n",
    "smooth_curr = jnp.arange(1, n_waypoints - 1)  # 1, 2, ..., n-2\n",
    "smooth_next = jnp.arange(2, n_waypoints)  # 2, 3, ..., n-1\n",
    "\n",
    "# Obstacle constraint: each waypoint x each obstacle.\n",
    "n_obstacles = len(obstacles)\n",
    "waypoint_ids_for_obs = jnp.tile(jnp.arange(n_waypoints), n_obstacles)\n",
    "obstacle_centers = jnp.array([[cx, cy] for cx, cy, r in obstacles])\n",
    "obstacle_radii = jnp.array([r for cx, cy, r in obstacles])\n",
    "# Repeat each obstacle for all waypoints.\n",
    "centers_repeated = jnp.repeat(obstacle_centers, n_waypoints, axis=0)\n",
    "radii_repeated = jnp.repeat(obstacle_radii, n_waypoints)\n",
    "\n",
    "# Build costs using batched construction.\n",
    "costs: list[jaxls.Cost] = [\n",
    "    # Anchor start and goal.\n",
    "    anchor_cost(WaypointVar(id=0), start),\n",
    "    anchor_cost(WaypointVar(id=n_waypoints - 1), goal),\n",
    "    # Smoothness costs (batched)\n",
    "    smoothness_cost(\n",
    "        WaypointVar(id=smooth_prev),\n",
    "        WaypointVar(id=smooth_curr),\n",
    "        WaypointVar(id=smooth_next),\n",
    "    ),\n",
    "    # Obstacle avoidance constraints (batched)\n",
    "    obstacle_constraint(\n",
    "        WaypointVar(id=waypoint_ids_for_obs),\n",
    "        centers_repeated,\n",
    "        radii_repeated,\n",
    "    ),\n",
    "]\n",
    "\n",
    "print(f\"Created {len(costs)} batched cost objects\")"
