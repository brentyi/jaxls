    "@jaxls.Cost.factory\n",
    "def spring_cost(\n",
    "    vals: jaxls.VarValues,\n",
    "    var_a: Point3Var,\n",
    "    var_b: Point3Var,\n",
    "    rest_length: jax.Array,\n",
    "    stiffness: float,\n",
    ") -> jax.Array:\n",
    "    \"\"\"Elastic potential energy for a Hookean spring.\n",
    "\n",
    "    Returns a 3D residual so squared cost is proportional to k(L - L0)^2.\n",
    "    \"\"\"\n",
    "    diff = vals[var_a] - vals[var_b]\n",
    "    length = jnp.sqrt(jnp.sum(diff**2) + 1e-6)\n",
    "    direction = diff / length\n",
    "    return (length - rest_length) * jnp.sqrt(stiffness) * direction\n",
    "\n",
    "\n",
    "@jaxls.Cost.factory\n",
    "def gravity_cost(\n",
    "    vals: jaxls.VarValues,\n",
    "    var: Point3Var,\n",
    "    mass: float,\n",
    "    g: float = 9.81,\n",
    ") -> jax.Array:\n",
    "    \"\"\"Gravitational potential energy mgh.\n",
    "\n",
    "    Returns sqrt(2mgh) so squared cost = 2mgh, giving constant\n",
    "    downward force mg independent of height.\n",
    "    \"\"\"\n",
    "    z = vals[var][2]\n",
    "    # softplus for numerical stability near z=0.\n",
    "    return jnp.sqrt(2.0 * mass * g * jax.nn.softplus(z) + 1e-8)\n",
    "\n",
    "\n",
    "@jaxls.Cost.factory(kind=\"constraint_geq_zero\")\n",
    "def table_constraint(\n",
    "    vals: jaxls.VarValues,\n",
    "    var: Point3Var,\n",
    ") -> jax.Array:\n",
    "    \"\"\"Inequality constraint: point must stay above the table (z >= 0).\"\"\"\n",
    "    return vals[var][2]\n",
    "\n",
    "\n",
    "@jaxls.Cost.factory(kind=\"constraint_geq_zero\")\n",
    "def sphere_constraint(\n",
    "    vals: jaxls.VarValues,\n",
    "    var: Point3Var,\n",
    "    center: jax.Array,\n",
    "    radius: float,\n",
    ") -> jax.Array:\n",
    "    \"\"\"Inequality constraint: point must stay outside the sphere.\n",
    "\n",
    "    Returns a scalar residual (distance constraints are inherently scalar).\n",
    "    \"\"\"\n",
    "    pos = vals[var]\n",
    "    dist_from_center = jnp.sqrt(jnp.sum((pos - center) ** 2) + 1e-8)\n",
    "    # Constraint: dist >= radius, i.e., dist - radius >= 0.\n",
    "    return dist_from_center - radius"
   ]
